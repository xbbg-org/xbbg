```python
import asyncio
import xbbg

async def fetch_prices(tickers):
    tasks = [xbbg.abdp(t, 'PX_LAST') for t in tickers]
    return await asyncio.gather(*tasks)

# Run concurrent requests
tickers = ['AAPL US Equity', 'MSFT US Equity', 'GOOGL US Equity']
results = asyncio.run(fetch_prices(tickers))
```
