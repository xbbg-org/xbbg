```python
import xbbg

# Simple streaming
async for batch in xbbg.astream(['AAPL US Equity'], ['LAST_PRICE', 'BID']):
    print(batch)

# Full control with subscription
sub = await xbbg.asubscribe(['AAPL US Equity'], ['LAST_PRICE'])
await sub.add(['MSFT US Equity'])  # Dynamic add
await sub.unsubscribe()
```
