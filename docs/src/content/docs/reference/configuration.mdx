```python
import xbbg
from xbbg import EngineConfig

# Configure before first request
xbbg.configure(
    request_pool_size=4,
    subscription_pool_size=8,
    host='localhost',
    port=8194,
)

# Or with EngineConfig
config = EngineConfig(request_pool_size=4)
xbbg.configure(config)
```
